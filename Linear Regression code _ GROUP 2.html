import pandas as pd
from sklearn.linear_model import LinearRegression
import matplotlib.pyplot as plt

# c file d liu CSV
data = pd.read_csv("TrafficTwoMonth.csv")

# Hin th 5 dng u tin  kim tra d liu
print("D liu:")
print(data.head())

# Chn cc bin u vo (features) v bin mc tiu (target)
X = data[['CarCount', 'BikeCount', 'BusCount', 'TruckCount']]  # Bin c lp
y = data['Total']  # Bin ph thuc (Total)

# Xy dng m hnh hi quy tuyn tnh
model = LinearRegression()
model.fit(X, y)

# Tnh cc gi tr d on (fitted values) v phn d (residuals)
fitted_values = model.predict(X)
residuals = y - fitted_values

# To biu  Histogram of Residuals
plt.figure(figsize=(10, 6))
plt.hist(residuals, bins=30, edgecolor='black', color='skyblue', alpha=0.7)
plt.title("Histogram of Residuals")
plt.xlabel("Residuals")
plt.ylabel("Frequency")
plt.grid(axis='y', linestyle='--', alpha=0.7)
plt.show()

import statsmodels.api as sm

# Chun b X v Y
X = data[bien_doc_lap]
Y = data[bien_phu_thuoc]

# Thm hng s vo X
X = sm.add_constant(X)

# Thc hin hi quy OLS
mo_hinh = sm.OLS(Y, X).fit()

# In kt qu hi quy
print(mo_hinh.summary())
